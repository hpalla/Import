public class CircleApp {

    public static void main(String[] args) {


        double radius = 10;
        double area;
        
        
        if (radius > 0) {

            area = calcArea(radius);

            System.out.println("A circle with radius " + radius
                    + " has an area of " + area);
        } 
        else {
            System.out.println("A circle with radius " + radius
                    + " is invalid. ");
        }

    }

    public static double calcArea(double r) {
        final double PI = 3.14;
        return PI * r * r;
    }
}
